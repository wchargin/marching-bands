% Usage: \puzzle{title}{clues for rows}{clues for bands}{solution}
\newcommand\puzzle[4]{%
  {\center\large\bfseries {#1}\endcenter}
  {\centering\marchingbands[0.75]\par\bigskip}
  \formatclues{#2}{#3}%
  \clearpage
  {\center\large\bfseries Solution to {#1}\endcenter}
  {\centering\marchingbandspuzzle[0.75]#4\endmarchingbandspuzzle\par\bigskip}
  \formatclues{#2}{#3}%
  \clearpage
}

% Usage: \formatclues{clues for rows}{clues for bands}
\newcommand\formatclues[2]{%
  \columnsep=18pt
  \begin{multicols}{3}%
    \raggedright
    \raggedcolumns
    \parindent=0pt
    \textbf{Rows}
    {\small\enumerate[label=\arabic*.]#1\endenumerate}
    \par\smallskip
    \textbf{Bands}
    {\small\enumerate[label=\Alph*.]#2\endenumerate}
  \end{multicols}
}
